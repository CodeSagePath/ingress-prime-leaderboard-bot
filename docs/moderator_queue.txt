import { useState } from "react";

type ModerationRequest = {
  id: string;
  displayName: string;
  submittedAt: string;
};

type ModeratorQueueProps = {
  initialRequests: ModerationRequest[];
};

export default function ModeratorQueue({ initialRequests }: ModeratorQueueProps) {
  const [requests, setRequests] = useState(initialRequests);
  const [loadingIds, setLoadingIds] = useState<string[]>([]);

  const handleAction = async (id: string, action: "approve" | "reject") => {
    setLoadingIds((prev) => [...prev, id]);
    try {
      await fetch(`/api/moderation/${action}`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ id }),
      });
      setRequests((prev) => prev.filter((request) => request.id !== id));
    } finally {
      setLoadingIds((prev) => prev.filter((loadingId) => loadingId !== id));
    }
  };

  return (
    <div className="space-y-4">
      {requests.length === 0 ? (
        <div className="rounded border border-dashed border-gray-300 p-6 text-center text-sm text-gray-500">
          No pending requests.
        </div>
      ) : (
        <ul className="space-y-3">
          {requests.map((request) => {
            const isLoading = loadingIds.includes(request.id);
            return (
              <li
                key={request.id}
                className="flex items-center justify-between rounded-lg border border-gray-200 bg-white p-4 shadow-sm"
              >
                <div>
                  <div className="text-sm font-semibold text-gray-900">{request.displayName}</div>
                  <div className="text-xs text-gray-500">{request.submittedAt}</div>
                </div>
                <div className="flex gap-2">
                  <button
                    onClick={() => handleAction(request.id, "reject")}
                    disabled={isLoading}
                    className="rounded border border-red-300 bg-white px-3 py-1 text-sm font-medium text-red-600 transition hover:bg-red-50 disabled:cursor-not-allowed disabled:opacity-50"
                  >
                    Reject
                  </button>
                  <button
                    onClick={() => handleAction(request.id, "approve")}
                    disabled={isLoading}
                    className="rounded bg-emerald-600 px-3 py-1 text-sm font-medium text-white transition hover:bg-emerald-700 disabled:cursor-not-allowed disabled:opacity-50"
                  >
                    Approve
                  </button>
                </div>
              </li>
            );
          })}
        </ul>
      )}
    </div>
  );
}
